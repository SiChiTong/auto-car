
## Compile as C++11, supported in ROS Kinetic and newer

  rospy
)

  ${PROJECT_SOURCE_DIR}
   ${catkin_INCLUDE_DIRS}

    mpc_solver.cc

    search.cc
    math_utils.cc)

     qp_solver)

    linear_quadratic_regulator.cc

    search.cc
